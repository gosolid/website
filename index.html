Solid

install

npm install -g @gosolid/solid

index.js

import express from 'express';

const app = express();

app.get('/', (req, res, next) => {
  res.send('Hello Solid!');
});

express.listen(3000, () => {
  console.info(' server listening on port 3000');
});

run

solid index.js
